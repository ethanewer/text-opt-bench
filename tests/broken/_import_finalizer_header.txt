# Prepended to a valid memory-task solution to test the import-time
# finalizer escape: an obfuscated __del__ that stops tracemalloc, on a
# cyclic object made unreachable at import time so it is collected at the
# evaluator's first gc point (before any peak/sample). Must be blocked —
# see tests/run_checks.py "import-time finalizer cannot zero the score".
import operator


class _Bomb:
    def __del__(self):
        try:
            g = operator.attrgetter("__globals__")(lambda: 0)
            g["__builtins__"]["__im" + "port__"]("tracemalloc").stop()
        except Exception:
            pass


def _arm():
    bombs = [_Bomb() for _ in range(30)]
    for b in bombs:
        b.self = b            # cyclic; unreachable once _arm() returns


_arm()
